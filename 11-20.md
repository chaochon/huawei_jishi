# 11. 记负均正
```
import java.util.Scanner;

public class Main{
    
    public static void main(String[] argv){
        
        Scanner in = new Scanner(System.in);
        while(in.hasNextInt()){
            int num = in.nextInt();
            int nagtiveCnt = 0;
            int postiveCnt = 0;
            double sum = 0.0;
            
            int nnum = 0;
            for(int i=0; i<num; i++){
                nnum = in.nextInt();
                if(nnum>0){
                    sum += nnum;
                    postiveCnt++;
                }else if(nnum<0)
                    nagtiveCnt++;
            }
            System.out.print(nagtiveCnt+" ");
            System.out.printf("%.1f\n", sum/(double)(postiveCnt));
        }
    }
}
```
